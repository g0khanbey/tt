# 1. IPBan'in koydugu ban kurallarini temizle
Get-NetFirewallRule -DisplayName "IPBan*" -ErrorAction SilentlyContinue | Remove-NetFirewallRule

# 2. IPBan calisiyorsa durdur
Stop-Process -Name "*IPBan*" -Force -ErrorAction SilentlyContinue
Stop-Service IPBan -Force -ErrorAction SilentlyContinue

# 3. Firewall'i gecici olarak kapat (RDP'yi geri almak icin)
Set-NetFirewallProfile -Profile Domain,Public,Private -Enabled False

# 4. RDP acik mi kontrol et
(Get-ItemProperty "HKLM:\System\CurrentControlSet\Control\Terminal Server").fDenyTSConnections
